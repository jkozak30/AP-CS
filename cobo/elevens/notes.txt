1.  public static String flip() {
        double ht = Math.random();
        if (ht > 2.0/3) {
            return "tails";
        } else {
            return "heads";
        }
    }
2.  public static boolean arePermutations(int[] one, int[] two) {
        for (int i=0; i<one.length; i++) {
            int thingCtr = 0;
            for (int j=0; j<one.length; j++) {
                if (one[i] == two[j]) {
                    thingCtr++;
                }
            }
            if (thingCtr == 0) { return false; }
        }
        return true;
    }
